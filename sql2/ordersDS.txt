SELECT CLASSICMODELS.ORDERS.STATUS,
CLASSICMODELS.ORDERS.ORDERDATE,
sum(CLASSICMODELS.ORDERDETAILS.QUANTITYORDERED *
CLASSICMODELS.ORDERDETAILS.PRICEEACH) as TotalPrice
FROM CLASSICMODELS.ORDERS, CLASSICMODELS.ORDERDETAILS
WHERE CLASSICMODELS.ORDERS.ORDERNUMBER = 
CLASSICMODELS.ORDERDETAILS.ORDERNUMBER
GROUP BY CLASSICMODELS.ORDERS.STATUS,
CLASSICMODELS.ORDERS.ORDERDATE
