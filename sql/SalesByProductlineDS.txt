SELECT CLASSICMODELS.PRODUCTS.PRODUCTLINE,
CLASSICMODELS.ORDERS.ORDERDATE,
CLASSICMODELS.ORDERDETAILS.PRICEEACH * CLASSICMODELS.ORDERDETAILS.QUANTITYORDERED as TotalSales FROM CLASSICMODELS.PRODUCTS,
CLASSICMODELS.ORDERS,
CLASSICMODELS.ORDERDETAILS
WHERE CLASSICMODELS.ORDERDETAILS.PRODUCTCODE =
CLASSICMODELS.PRODUCTS.PRODUCTCODE
and CLASSICMODELS.ORDERS.ORDERNUMBER =
CLASSICMODELS.ORDERDETAILS.ORDERNUMBER
